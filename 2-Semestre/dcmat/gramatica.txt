
Calc -> QUIT EOL
        | Comandos EOL

Comandos -> SHOW Show SEMICOLON
            | SET Set SEMICOLON
            | ExpressionAddictive
            | ExpressionMultiplicative

Show -> SYMBOLS 
        | MATRIX
        | SETTINGS

Set -> FLOAT PRECISION INT SEMICOLON 
    | H_VIEW  COLON Limit SEMICOLON 
    | V_VIEW NumValue COLON NumValue SEMICOLON 
    | AXIS BoolValue SEMICOLON 
    | ERASE PLOT BoolValue SEMICOLON 
    | INTEGRAL_STEPS INT SEMICOLON 

ExpressionAddictive -> ExpressionMultiplicative ADD ExpressionAddictive
                       | ExpressionMultiplicative SUB ExpressionAddictive
                       | ExpressionMultiplicative

ExpressionMultiplicative -> ExpressionPotentiation MULTIPLY ExpressionMultiplicative
                            | ExpressionPotentiation DIV ExpressionMultiplicative
                            | ExpressionPotentiation
                        
ExpressionPotentiation -> IsNegative POW ExpressionPotentiation
                          | IsNegative REM ExpressionPotentiation
                          | IsNegative

IsNegative -> ADD Term
             | SUB Term
             | Term

Term -> ID
        | NumValue
        | SEN Delimted
        | COS Delimted
        | TAN Delimted
        | ABS Delimted
        | Delimted 

Delimted -> L_PAREN ExpressionAddictive R_PAREN
