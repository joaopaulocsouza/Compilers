Dcmat -> Command EOL
        | Att EOL
        | QUIT EOL
        | EOL

Command -> SET SetLinha
           | SHOW ShowLinha
           | RESET SETTINGS SEMICOLON
           | PLOT SEMICOLON
           | PLOT L_PAREN Funcao R_PAREN SEMICOLON
           | RPN L_PAREN Expressao R_PAREN SEMICOLON
           | INTEGRATE L_PAREN Value COLON VALUE COMMA Funcao R_PAREN SEMICOLON
           | SUM L_PAREN VAR COMMA Value COLON Value COMMA Expressao R_PAREN SEMICOLON
           | MATRIX EQUAL L_SQUARE_BRACKET L_SQUARE_BRACKET Value MatrixLinha R_SQUARE_BRACKET R_SQUARE_BRACKET SEMICOLON
           | SOLVE SolveLinha

MatrixEstrela -> COMMA L_SQUARE_BRACKET Value MatrixLinha R_SQUARE_BRACKET
                 | ɛ

MatrixLinha -> COMMA Value MatrixLinha
               | ɛ
            
SolveLinha -> DETERMINANT SEMICOLON
              | LINEAR_SYSTEM SEMICOLON

ShowLinha -> SETTINGS SEMICOLON
            | MATRIX SEMICOLON

SetLinha -> H_VIEW Value COLON Value SEMICOLON
           | V_VIEW Value COLON Value SEMICOLON
           | AXIS Bool SEMICOLON
           | ERASE PLOT Bool SEMICOLON
           | INTEGRAL_STEPS INT SEMICOLON

Att: IDENTIFIER ASSIGN Expressao SEMICOLON

Bool -> ON | OFF

Value -> INT | REAL

Funcao: SEN 
        | COS
        | TAN
        | ABS

Expressao ->  Termo ExpressaoLinha;
              | + Termo ExpressaoLinha;
              | - Termo ExpressaoLinha;
              | * Termo ExpressaoLinha;
              | / Termo ExpressaoLinha;
              | % Termo ExpressaoLinha;
              

ExpressaoLinha -> + Fator ExpressaoLinha;
              | - Fator ExpressaoLinha;
              | * Fator ExpressaoLinha;
              | / Fator ExpressaoLinha;
              | % Fator ExpressaoLinha;
              | ;

Fator -> ( Expressao )
         | INTEIRO
         | REAL
         | IDENTIFICADOR

Termo -> INT
        | FLOAT
        | IDENTIFIER
